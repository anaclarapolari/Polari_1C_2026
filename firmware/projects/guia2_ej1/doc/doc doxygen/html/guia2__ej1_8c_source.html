137static void KeyReadTask(void *pvParameter) {
138 while (true) {
139 // Leer estado actual de los switches
140 int8_t current_switches = SwitchesRead();
141
142 // Detectar presiones (cambio de 0 a 1)
143 int8_t pressed_switches = current_switches & ~last_switches; // lo hago para resolver el problema de seleccionar por mucho tiempo y que pase más de 10ms
144
145 // Procesar TEC1 (SWITCH_1)
146 if (pressed_switches & SWITCH_1) {
147 medida = !medida;
148 }
149
150 // Procesar TEC2 (SWITCH_2)
151 if (pressed_switches & SWITCH_2) {
152 if (medida) {
153 hold_mode = !hold_mode;
154 }
155 }
156
157 // Guardar estado actual para la próxima lectura
158 last_switches = current_switches;
159
160 // Esperar 10 milisegundos
161 vTaskDelay(KEY_READ_PERIOD_MS / portTICK_PERIOD_MS);
162 }
163}

168static void MeasurementTask(void *pvParameter) {
169 while (true) {
170 if (medida) {
171 // SIEMPRE leer la distancia (incluso en HOLD)
172 last_distance = HcSr04ReadDistanceInCentimeters();
173
174 // SIEMPRE controlar los LEDs según la medida actual
175 ControlLedsBasedOnDistance(last_distance);
176
177 // Actualizar LCD según el modo
178 if (!hold_mode) {
179 // Modo normal: mostrar la medida actual en LCD
180 if (last_distance <= 999) {
181 LcdItsE0803Write(last_distance);
182 }
183 printf("Distancia: %u cm\n", last_distance);
184 }
185 } else {
186 // Cuando no está midiendo, apagar todo
187 LedsOffAll();
188 LcdItsE0803Off();
189 }
190
191 // Esperar 1 segundo
192 vTaskDelay(MEASUREMENT_PERIOD_MS / portTICK_PERIOD_MS);
193 }
194}

196static void ControlLedsBasedOnDistance(uint16_t distance) {
197 if (distance < DISTANCE_THRESHOLD_1) {
198 // Distancia < 10 cm: Apagar todos los LEDs
199 LedsOffAll();
200 } else if (distance < DISTANCE_THRESHOLD_2) {
201 // Distancia 10-20 cm: Encender solo LED_1
202 LedsOffAll();
203 LedOn(LED_1);
204 } else if (distance < DISTANCE_THRESHOLD_3) {
205 // Distancia 20-30 cm: Encender LED_1 y LED_2
206 LedsOffAll();
207 LedOn(LED_1);
208 LedOn(LED_2);
209 } else {
210 // Distancia >= 30 cm: Encender LED_1, LED_2 y LED_3
211 LedOn(LED_1);
212 LedOn(LED_2);
213 LedOn(LED_3);
214 }
215}

224void app_main(void) {
225
226 // Inicializar LEDs
227 if (!LedsInit()) {
228 return;
229 }
230
231 // Inicializar sensor HC-SR04
232 if (!HcSr04Init(HC_SR04_ECHO_GPIO, HC_SR04_TRIGGER_GPIO)) {
233 LedsOffAll();
234 return;
235 }
236
237 // Inicializar display LCD
238 if (!LcdItsE0803Init()) {
239 LedsOffAll();
240 HcSr04Deinit();
241 return;
242 }
243 // Inicializar switches
244 if (SwitchesInit() == -1) {
245 LedsOffAll();
246 HcSr04Deinit();
247 LcdItsE0803DeInit();
248 return;
249 }
250
251 // Crear tarea de lectura de teclas (cada 10 ms)
252 xTaskCreate(&KeyReadTask, "KeyReadTask", 1024, NULL, 5, &key_read_task_handle);
253
254 // Crear tarea de medición (cada 1 segundo)
255 xTaskCreate(&MeasurementTask, "MeasurementTask", 2048, NULL, 5, &measurement_task_handle);
256}
