132static void KeyReadTask(void *pvParameter) {
133 while (true) {
134 int8_t current_switches = SwitchesRead(); // 00000001 si apretas TEC1
135
136 int8_t pressed_switches = current_switches & ~last_switches; // lo hago para resolver el problema de seleccionar por mucho tiempo y que pase más de 10ms
137 // 00000001 & 11111110 = 00000000 -> no detecta nada
138 // Procesar TEC1 (SWITCH_1): deja de medir o vuelve a medir
139 if (pressed_switches & SWITCH_1) {
140 medida = !medida;
141 }
142
143 // Procesar TEC2 (SWITCH_2): congela el display o lo descongela
144 if (pressed_switches & SWITCH_2) {
145 if (medida) {
146 hold_mode = !hold_mode;
147 }
148 }
149
150 last_switches = current_switches;
151
152 vTaskDelay(KEY_READ_PERIOD_MS / portTICK_PERIOD_MS);
153 }
154}

156static void MeasurementTask(void *pvParameter) {
157 while (true) {
158 if (medida) {
159 last_distance = HcSr04ReadDistanceInCentimeters(); // lee la distancia
160 ControlLedsBasedOnDistance(last_distance); // compara la distancia y enciende leds
161
162 if (!hold_mode) {
163 // Modo normal: mostrar la medida actual en LCD
164 if (last_distance <= 999) {
165 LcdItsE0803Write(last_distance);
166 }
167 }
168 } else {
169 // Cuando no está midiendo, apagar todo
170 LedsOffAll();
171 LcdItsE0803Off();
172 }
173
174 vTaskDelay(MEASUREMENT_PERIOD_MS / portTICK_PERIOD_MS);
175 }
176}

178static void ControlLedsBasedOnDistance(uint16_t distance) {
179 if (distance < DISTANCE_THRESHOLD_1) {
180 // Distancia < 10 cm: Apagar todos los LEDs
181 LedsOffAll();
182 } else if (distance < DISTANCE_THRESHOLD_2) {
183 // Distancia 10-20 cm: Encender solo LED_1
184 LedsOffAll();
185 LedOn(LED_1);
186 } else if (distance < DISTANCE_THRESHOLD_3) {
187 // Distancia 20-30 cm: Encender LED_1 y LED_2
188 LedsOffAll();
189 LedOn(LED_1);
190 LedOn(LED_2);
191 } else {
192 // Distancia >= 30 cm: Encender LED_1, LED_2 y LED_3
193 LedOn(LED_1);
194 LedOn(LED_2);
195 LedOn(LED_3);
196 }
197}

203void app_main(void) {
204
205 // Inicializar LEDs
206 if (!LedsInit()) {
207 return;
208 }
209
210 // Inicializar sensor HC-SR04
211 if (!HcSr04Init(HC_SR04_ECHO_GPIO, HC_SR04_TRIGGER_GPIO)) {
212 LedsOffAll();
213 return;
214 }
215
216 // Inicializar display LCD
217 if (!LcdItsE0803Init()) {
218 LedsOffAll();
219 HcSr04Deinit();
220 return;
221 }
222 // Inicializar switches
223 if (SwitchesInit() == -1) {
224 LedsOffAll();
225 HcSr04Deinit();
226 LcdItsE0803DeInit();
227 return;
228 }
229
230 // Crear tarea de lectura de teclas (cada 10 ms)
231 // paso la función, el nombre de la tarea, el espacio en memoria, parámetros, prioridad y ID
232 xTaskCreate(&KeyReadTask, "KeyReadTask", 1024, NULL, 5, &key_read_task_handle);
233
234 // Crear tarea de medición (cada 1 segundo)
235 xTaskCreate(&MeasurementTask, "MeasurementTask", 2048, NULL, 5, &measurement_task_handle);
236}
