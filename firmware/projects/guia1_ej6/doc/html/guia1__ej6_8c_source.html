114int8_t convertToBcdArray(uint32_t data, uint8_t digits, uint8_t *bcd_number)
115{
116 if (bcd_number == NULL || digits == 0) {
117 return -1;
118 }
119 for (uint8_t i = 0; i < digits; i++) {
120 bcd_number[i] = 0;
121 }
122 for (int i = digits - 1; i >= 0; i--) {
123 bcd_number[i] = data % 10;
124 data /= 10;
125 if (data == 0) {
126 break;
127 }
128 }
129 if (data != 0) {
130 return -2;
131 }
132 return 0;
133}

152void setBCDToGPIOs(uint8_t bcdDigit, gpioConf_t *gpioConfigs)
153{
154 for (int i = 0; i < 4; i++) {
155 GPIOInit(gpioConfigs[i].pin, gpioConfigs[i].dir);
156 }
157 for (int i = 0; i < 4; i++) {
158 bool bitState = (bcdDigit & (1 << i)) != 0;
159 GPIOState(gpioConfigs[i].pin, bitState);
160 }
161}

184void displayValueOnDigits(uint32_t data, uint8_t digits, gpioConf_t *mapBcd, gpioConf_t *mapDigits)
185{
186 if (digits == 0 || digits > 3) {
187 return;
188 }
189 // Inicializa los GPIOs de dígito (desactivados inicialmente)
190 for (int d = 0; d < digits; d++) {
191 GPIOInit(mapDigits[d].pin, mapDigits[d].dir);
192 GPIOState(mapDigits[d].pin, 0);
193 }
194
195 // Convertir data a arreglo BCD
196 uint8_t bcd_digits[3] = {0};
197 if (convertToBcdArray(data, digits, bcd_digits) != 0) {
198 return; // Error en conversión
199 }
200
201 // Para cada ciclo de refresco, mostramos un dígito a la vez
202 for (uint8_t pos = 0; pos < digits; pos++) {
203 uint8_t digitValue = bcd_digits[pos];
204
205 // Seteo BCD del dígito actual
206 setBCDToGPIOs(digitValue, mapBcd);
207
208 // Activar el dígito correspondiente del display
209 GPIOState(mapDigits[pos].pin, true);
210
211 // Apagar el dígito previo
212 GPIOState(mapDigits[pos].pin, false);
213 }
214}
