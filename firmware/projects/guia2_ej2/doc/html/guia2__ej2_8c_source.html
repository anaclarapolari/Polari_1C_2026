89static void MedirDistancia(void *param) {
90 if (medida) {
91 // SIEMPRE leer la distancia (incluso en HOLD)
92 last_distance = HcSr04ReadDistanceInCentimeters();

107static void ControlLedsBasedOnDistance(uint16_t distance) {
108 if (distance < DISTANCE_THRESHOLD_1) {
109 // Distancia < 10 cm: Apagar todos los LEDs
110 LedsOffAll();
111 } else if (distance < DISTANCE_THRESHOLD_2) {
112 // Distancia 10-20 cm: Encender solo LED_1
113 LedsOffAll();
114 LedOn(LED_1);
115 } else if (distance < DISTANCE_THRESHOLD_3) {
116 // Distancia 20-30 cm: Encender LED_1 y LED_2
117 LedsOffAll();
118 LedOn(LED_1);
119 LedOn(LED_2);
120 } else {
121 // Distancia >= 30 cm: Encender LED_1, LED_2 y LED_3
122 LedsOffAll();
123 LedOn(LED_1);
124 LedOn(LED_2);
125 LedOn(LED_3);
126 }
127}

133void app_main(void) {
134
135 // Inicializar LEDs
136 if (!LedsInit()) {
137 return;
138 }
139
140 // Inicializar sensor HC-SR04
141 if (!HcSr04Init(HC_SR04_ECHO_GPIO, HC_SR04_TRIGGER_GPIO)) {
142 LedsOffAll();
143 return;
144 }
145
146 // Inicializar display LCD
147 if (!LcdItsE0803Init()) {
148 LedsOffAll();
149 HcSr04Deinit();
150 return;
151 }
152 // Inicializar switches
153 if (SwitchesInit() == -1) {
154 LedsOffAll();
155 HcSr04Deinit();
156 LcdItsE0803DeInit();
157 return;
158 }
159
160 // configuro interrupciones: paso la tecla, la función a ejecutar y argumentos no hay
161 // reemplazo la tarea que leía las teclas cada 10 ms
162 SwitchActivInt(SWITCH_1, InterrupcionTEC1, NULL);
163 SwitchActivInt(SWITCH_2, InterrupcionTEC2, NULL);
164
165 // configuro timer cada 1 segundo. Cada un segundo se mide la distancia
166 timer_config_t timer_config = {
167 .timer = TIMER_A,
168 .period = MEASUREMENT_PERIOD_US,
169 .func_p = MedirDistancia,
170 .param_p = NULL
171 };
172 TimerInit(&timer_config); //le paso la dirección de memoria del timer que configuré recién
173 TimerStart(TIMER_A);
174}
